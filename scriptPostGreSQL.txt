CREATE DATABASE agenda
  WITH OWNER = postgres
       ENCODING = 'UTF8'
       TABLESPACE = pg_default
       LC_COLLATE = 'Portuguese_Brazil.1252'
       LC_CTYPE = 'Portuguese_Brazil.1252'
       CONNECTION LIMIT = -1;



CREATE TABLE public.agenda_telefonica
(
  id integer NOT NULL DEFAULT nextval('agenda_telefonica_id_seq'::regclass),
  nome text NOT NULL,
  telefone text NOT NULL,
  CONSTRAINT agenda_telefonica_pkey PRIMARY KEY (id)
)
WITH (
  OIDS=FALSE
);
ALTER TABLE public.agenda_telefonica
  OWNER TO postgres;
